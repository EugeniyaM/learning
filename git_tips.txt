cmd - коммандная строка

cd - указание директории
cls/clear- очистить экран коммандной строки
mkdir имя_каталога - создать каталог
rmdir имя_каталога - удалить каталог

git init - создать репозиторий
git add <file> - добавить файл в репозиторий,добавить изменения, индексирование
git add . - проиндексировать ВСЕ файлы (прверять статус перед использованием)
git restore --staged <file> - отменяет добавление изменений


git commit -m "First Commit" - коммит файла + задать имя коммиту (фиксация изменений),
	-m - комментарий в коммендной строке
<!-- Чтобы сделать коммит для нескольких файлов:
	git add a.html
	git add b.html
	git commit -m "Changes for a and b" -->

git status - текущее состояние репозитория (где находимся, нужно ли коммитить)
git push - отправить изменения на сервер
git checkout -b + имя ветки - создать ветку
git ls-remote origin - 
git push -u origin имя_ветки- отправить ветку на гитхаб
git merge имя ветки - объединение веток (из уровня выше на уровень ниже)
git pull - подтянуть все изменения с сервера
